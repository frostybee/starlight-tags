import { TagsList } from 'starlight-tags/components';

This page demonstrates the `TagsList` component, which renders a collection of tags with customizable layouts and sorting.

## Basic Usage

Display all tags in the system:

<TagsList limit={8} />

```astro
import { TagsList } from 'starlight-tags/components';

<TagsList limit={8} />
```

## Layouts

### Inline (Default)

Tags flow horizontally and wrap:

<TagsList layout="inline" limit={6} />

```astro
<TagsList layout="inline" />
```

### Grid

Tags in a responsive grid:

<TagsList layout="grid" limit={6} />

```astro
<TagsList layout="grid" />
```

### List

Tags stacked vertically:

<TagsList layout="list" limit={6} />

```astro
<TagsList layout="list" />
```

## With Title

Add a heading above the tags:

<TagsList title="Related Topics" limit={5} />

```astro
<TagsList title="Related Topics" />
```

## Style Variants

### Solid (Default)

<TagsList variant="solid" limit={5} />

### Outline

<TagsList variant="outline" limit={5} />

### Ghost

<TagsList variant="ghost" limit={5} />

## With Page Counts

Show how many pages use each tag:

<TagsList showCount={true} limit={6} />

```astro
<TagsList showCount />
```

## Specific Tags

Display only specific tags by ID:

<TagsList tagIds={['configuration', 'guide', 'plugins', 'advanced']} />

```astro
<TagsList tagIds={['configuration', 'guide', 'plugins', 'advanced']} />
```

## Grouped Examples

### Content Types

<TagsList tagIds={['guide', 'reference', 'tutorial', 'example']} title="Content Types" variant="outline" />

### Audience

<TagsList tagIds={['getting-started', 'advanced', 'migration', 'troubleshooting']} title="Audience" layout="grid" />

### Integrations

<TagsList tagIds={['plugins', 'frameworks', 'deployment']} title="Integrations" showCount={true} />

## Sorting

### By Name (Default)

Sort tags alphabetically:

<TagsList sortBy="name" limit={6} />

```astro
<TagsList sortBy="name" />
```

### By Count

Sort by number of pages using each tag (highest first):

<TagsList sortBy="count" limit={6} showCount={true} />

```astro
<TagsList sortBy="count" showCount />
```

## Combined Options

Grid layout with title, counts, and outline variant:

<TagsList
  title="Popular Topics"
  layout="grid"
  showCount={true}
  variant="outline"
  limit={6}
/>

```astro
<TagsList
  title="Popular Topics"
  layout="grid"
  showCount
  variant="outline"
  limit={6}
/>
```
